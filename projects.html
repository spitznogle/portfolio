Animal Manager Console App

Interactive Python console app with modular functions and loops.


from functions import add_animal, list_animals

animals = []

# Menu Loop
while True:
    print("1. Add Animal")
    print("2. List Animals")
    print("3. Exit")
    choice = input("Enter choice: ")

    if choice == "1":
        add_animal(animals)
    elif choice == "2":
        list_animals(animals)
    elif choice == "3":
        break
    else:
        print("Invalid choice.")
        

# Ask user to input animal information
animal_species = input("Enter animal's species: ")
animal_name = input(f"Enter the {animal_species}'s name: ")
animal_age = int(input(f"Enter {animal_name}'s age: "))

# Age Classification
if animal_age < 1:
    age_category = "Baby"
elif animal_age <= 4:
    age_category = "Young"
elif animal_age <= 7:
    age_category = "Adult"
else:
    age_category = "Senior"

# Conditional Logic for Adoption Status
adopted_input = input("Has this animal been adopted? (yes/no): ").lower()
if adopted_input == "yes":
    adopted = True
else:
    adopted = False

print("\nYou just added a new animal!")
print(f"Name: {animal_name}")
print(f"Species: {animal_species}")
print(f"Age: {animal_age}")
print(f"{animal_name} is classified as a {age_category}")

print(f"Adopted? {adopted}")
