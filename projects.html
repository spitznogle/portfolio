Animal Manager Console App

Interactive Python console app with modular functions and loops.


from functions import add_animal, list_animals

animals = []

# Menu Loop
while True:
    print("1. Add Animal")
    print("2. List Animals")
    print("3. Exit")
    choice = input("Enter choice: ")

    if choice == "1":
        add_animal(animals)
    elif choice == "2":
        list_animals(animals)
    elif choice == "3":
        break
    else:
        print("Invalid choice.")
        

def add_animal(animals_list):
    """
    Adds a new animal to the shelter list.
    Prompts the user for name, species, age, and adoption status.
    """
    name = input("Enter animal's name: ")
    species = input("Enter species (Dog/Cat): ")
    age = int(input("Enter age: "))
    adopted_input = input("Is the animal adopted? (yes/no): ").lower()
    adopted = adopted_input == "yes"

    animal = {
        "name": name,
        "species": species,
        "age": age,
        "adopted": adopted
    }

    animals_list.append(animal)
    print(f"{name} has been added to the shelter.")


def list_animals(animals_list):
    """
    Prints all animals that are currently in the shelter
    and not adopted yet.
    """
    print("Current Animals in Shelter:")
    for animal in animals_list:
        if not animal["adopted"]:
            print(f"{animal['name']}, {animal['species']}, Age: {animal['age']}")
